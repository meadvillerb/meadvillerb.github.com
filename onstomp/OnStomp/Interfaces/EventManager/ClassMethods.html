def create_event_method name
  event_methods << name
  module_eval "def #{name}(&block); bind_event(:#{name}, block); end"
end

def create_event_methods name, *prefixes
  prefixes << :on if prefixes.empty?
  prefixes.each { |pre| create_event_method :#{pre}_#{name}" }
end
