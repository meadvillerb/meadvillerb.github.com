def open(*args)
  client = OnStomp::Client.new(self, *args)
  client.extend OnStomp::OpenURI::ClientExtensions
  client.auto_destination = self.path
  if block_given?
    begin
      client.connect
      yield client
    ensure
      client.disconnect
    end
  end
  client
end
