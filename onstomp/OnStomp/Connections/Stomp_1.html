def unsubscribe_frame f, h
  id = f.is_a?(OnStomp::Components::Frame) ? f[:id] : f
  create_frame('UNSUBSCRIBE', [{:id => id}, h]).tap do |f|
    raise ArgumentError, 'subscription ID could not be determined' unless f.header?(:id)
  end
end
